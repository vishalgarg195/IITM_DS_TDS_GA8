This report loads employee data, calculates the frequency of the Finance department, prints the result, and displays a histogram of department distribution.

Python Code Used


import pandas as pd
import matplotlib.pyplot as plt

# Employee dataset (100 synthetic rows)
data = [
    ["EMP001","Finance","Europe",70.13,10,4],
    ["EMP002","Operations","Europe",61.98,10,3.9],
    ["EMP003","R&D","Africa",90.23,4,4.9],
    ["EMP004","IT","Asia Pacific",68.48,9,4.8],
    ["EMP005","HR","Middle East",60.35,1,3.5]
]

# Extend dataset to 100 rows
import random
departments = ["Finance","HR","IT","R&D","Operations","Sales","Marketing"]
regions = ["Europe","Africa","Asia Pacific","Latin America","Middle East","North America"]

for i in range(6,101):
    data.append([
        f"EMP{str(i).zfill(3)}",
        random.choice(departments),
        random.choice(regions),
        round(random.uniform(55,95),2),
        random.randint(1,15),
        round(random.uniform(3.0,5.0),1)
    ])

df = pd.DataFrame(data, columns=[
    "employee_id","department","region","performance_score","years_experience","satisfaction_rating"
])

# Count Finance employees
finance_count = df[df["department"] == "Finance"].shape[0]
print("Finance Department Count:", finance_count)

# Histogram
plt.figure(figsize=(8,6))
plt.hist(df["department"], edgecolor="black")
plt.title("Department Distribution")
plt.xlabel("Department")
plt.ylabel("Frequency")

plt.savefig("hist.png")
